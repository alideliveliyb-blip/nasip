        { id: 12, g: "Jüpiter", t: "Bilinç", k: [1, 2, 3], p: "Jupiter" }

    ];


    let userResults = [];


    function show(s) {

        document.querySelectorAll('.screen').forEach(e => e.classList.remove('active'));

        document.getElementById('sc' + s).classList.add('active');

    }


    function runAnalysis() {

        const name = document.getElementById('uName').value;

        const date = document.getElementById('uDate').value;

        if(!name || !date) return alert("Bilgileri eksiksiz giriniz.");


        const kod = (name.length % 9) || 9;

        document.getElementById('zodiacInfo').innerText = name + ", Burcunuzun Asaletiyle Hoş Geldiniz.";

        

        let nodes = document.getElementById('nodes');

        nodes.innerHTML = "";

        userResults = [];


        systemData.forEach((h, i) => {

            let d = new Date(date);

            d.setMonth(d.getMonth() + i);

            let gun = (d.getDate() + (d.getMonth() * 3)) % 30 || 30; // KBA Progresyonu

            let evreIdx = Math.ceil(gun / 10) - 1;

            let evreKod = h.k[evreIdx];

            

            let status = evreKod === 1 ? "OLUR" : (evreKod === 2 ? "SIKINTILI" : "OLMAZ");

            if (kod === 2 && h.id === 10) status = "İSMI AZAM";


            userResults.push({ ...h, status, gun });

            nodes.innerHTML += `<div class="house-node" onclick="viewHouse(${i})">${i+1}. EV<br><small>${h.t}</small></div>`;

        });

        show(4);

    }


    function viewHouse(i) {

        const data = userResults[i];

        const planetImgQuery = ``;

        let cls = data.status === "OLUR" ? "bg-olur" : (data.status === "SIKINTILI" ? "bg-sikinti" : (data.status === "OLMAZ" ? "bg-olmaz" : "bg-azam"));

        

        let info = data.status === "OLUR" ? "İrade ve zaman bir olmuş. Doğal başarı." : 

                   data.status === "SIKINTILI" ? "Yaşam alanı düzenleme bekliyor. Dirayet onarılmalı." : 

                   data.status === "İSMI AZAM" ? "MUTLAK KİLİT: Bu ev en büyük tekamül sınavındır." : "BİLE BİLE ZAAF: Nefsi tercihlerine uyanık ol.";


        document.getElementById('details').innerHTML = `

            <div class="detail-view">

                <div style="flex:1">

                    <h3>${i+1}. Ev: ${data.t}</h3>

                    <p>Yönetici: <strong>${data.g}</strong></p>

                    <p>Frekans Durumu: <span class="badge ${cls}">${data.status}</span></p>

                    <p style="font-style:italic; color:#aaa;">${info}</p>

                </div>

                <div style="text-align:center;">

                    ${planetImgQuery}<br>

                    <small style="color:var(--gold)">${data.g} Enerjisi</small>

                </div>

            </div>`;

    }


    function showSummary() {

        let sumHtml = "<ul>";

        userResults.forEach(r => {

            if(r.status !== "OLUR") {

                sumHtml += `<li><strong>${r.t} Sahası:</strong> ${r.status} - İtidal ile onarılmalı.</li>`;

            }

        });

        sumHtml += "</ul>";

        document.getElementById('finalSummary').innerHTML = sumHtml;

    }
